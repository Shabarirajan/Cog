Function Main
    Declare Integer F0, F1, F2, F3, FN, N, Count
    
    Output "The value of the Nth term fibonacci number is"
    Input N
    Assign F0 = 0
    Assign F1 = 1
    Assign F2 = 1
    Assign F3 = 2
    Assign Count = 0
    While Count < N
        Output F1
        Assign FN = F1+F2
        Assign F1 = F2
        Assign F2 = FN
        Assign Count = Count + 1
    End
End
